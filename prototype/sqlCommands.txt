/* ******************** Create ******************* */
CREATE TABLE outcome (id int(100) NOT NULL AUTO_INCREMENT, presentation double NOT NULL, advertisement double NOT NULL, management double NOT NULL, employee_salary double NOT NULL, chancery double NOT NULL, mail_and_shipping double NOT NULL, transportation double NOT NULL, assignment double NOT NULL, other double NOT NULL, total double NOT NULL, PRIMARY KEY (id));
CREATE TABLE income (id int(100) NOT NULL AUTO_INCREMENT, from_inside double NOT NULL, from_people double NOT NULL, other_parties double NOT NULL, other double NOT NULL, total double NOT NULL, PRIMARY KEY (id));
CREATE TABLE charity (id int(100) NOT NULL AUTO_INCREMENT, `from` varchar(100) NOT NULL, `to` varchar(100) NOT NULL, total double NOT NULL, PRIMARY KEY (id));
CREATE TABLE finance (id int(10) NOT NULL AUTO_INCREMENT, charityid int(100) NOT NULL, incomeid int(100) NOT NULL, outcomeid int(100) NOT NULL, year varchar(50) NOT NULL, month varchar(2) NOT NULL, day varchar(3) NOT NULL, name varchar(100), donation double, remaining double NOT NULL, PRIMARY KEY (id), INDEX (id));
CREATE TABLE people_aan (id int(100) NOT NULL AUTO_INCREMENT, partycode int(100) NOT NULL, candidateid int(100) NOT NULL, year int(100) NOT NULL, type varchar(50) NOT NULL, `size` double NOT NULL, PRIMARY KEY (id));
CREATE TABLE candidate (id int(100) NOT NULL AUTO_INCREMENT, partycode int(100) NOT NULL, firstname varchar(100) NOT NULL, lastname varchar(100) NOT NULL, PRIMARY KEY (id));
CREATE TABLE party (code int(100) NOT NULL AUTO_INCREMENT, title varchar(100) NOT NULL, acronym varchar(50), financeid int(10) NOT NULL, PRIMARY KEY (code), INDEX (code));
